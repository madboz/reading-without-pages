let pageTitle = document.title;
let titlesArray = [];

const retrievedData = localStorage.getItem("titles");
if (retrievedData != null)
{
    titlesArray.push(JSON.parse(retrievedData));   
}

titlesArray.push(pageTitle);

localStorage.setItem("titles", JSON.stringify(titlesArray));

let newStr ="";
titlesArray.forEach(function (item) {
    newStr += item + " - ";
});

let bar = document.getElementById("bar-text");
bar.innerHTML = newStr;
