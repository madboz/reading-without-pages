
// IE does not know about the target attribute. It looks for srcElement
// This function will get the event target in a browser-compatible way


function getEventTarget(e) {
	e = e || window.event;
	return e.target || e.srcElement;
};





// THis function will get executed when we click on any left side TOPIC

let ul = document.getElementById('LeftTopic');
ul.onclick = function (event) {
	let target = getEventTarget(event);
	console.log("Left Side Selected Topic " + target.innerHTML);
	let articles, topic;
	let selected_topic = target.innerHTML;


	articles = document.querySelectorAll(".article");

	for (let i = 0; i < articles.length; i++) {
		topic = articles[i].dataset.topic;
		console.log("Current Topic : " + topic);
		if (topic.toLowerCase() !== selected_topic.toLowerCase()) {
			articles[i].style.display = "none";

		} else {
			articles[i].style.display = "inline-block";
		}
	}
};
